  cd /tmp && \
  curl -Lo scala-${SCALA_VERSION}.tgz https://downloads.typesafe.com/scala/${SCALA_VERSION}/scala-${SCALA_VERSION}.tgz && \
  tar xf scala-${SCALA_VERSION}.tgz && \
  mv scala-${SCALA_VERSION} ${SCALA_HOME} && \
  rm scala-${SCALA_VERSION}.tgz
